#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[] )
{
 int i;
 long sum = 0;
 if (argc != 0)
 {
 for (i = 0; i < argc; i++)
 {
 sum +=  atoi (argv[i]);
 }
 printf("\n%d", sum);
 } 
 return 0;
}
